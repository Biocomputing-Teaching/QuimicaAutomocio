\begin{exr}{Entalpia de reacció}
Tenint en compte aquestes energies d'enllaç:
\begin{center}
\begin{tabular}{cc}
& $E_b$ / kJ mol$^{-1}$ \\
\hline
C-O al monòxid de carboni & +1077 \\
C-O al diòxid de carboni & +805 \\
O-H & +464 \\
H-H & +436 \\
\hline
\end{tabular}
\end{center}

Calcula l'entalpia de la reacció:
\ch{CO_{(g)} + H2O_{(g)} -> CO2_{(g)} + H2_{(g)}} 
\end{exr}
